function error takes string msg returns nothing
	local integer hash
	if MagicFunctions_compiletime then
		call print("ERROR: " + msg + "\n" + "")
	else
		if  not ErrorHandling_suppressErrorMessages then
			set hash = string_getHash(msg)
			if hashtable_hasInt(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash) then
				if hashtable_loadInt(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash) + ErrorHandling_MUTE_ERROR_DURATION < GameTimer_currentTime then
					call Log_error(msg + "")
					call hashtable_saveInt(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash, real_toInt(GameTimer_currentTime))
					call hashtable_saveBoolean(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash, false)
				elseif hashtable_hasBoolean(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash) then
					if  not hashtable_loadBoolean(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash) then
						call Log_error("|cffFF3A29Excessive repeating errors are being omitted")
						call hashtable_saveBoolean(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash, true)
					endif
				else
					call Log_error("|cffFF3A29Excessive repeating errors are being omitted")
					call hashtable_saveBoolean(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash, true)
				endif
			else
				call hashtable_saveInt(ErrorHandling_HT, ErrorHandling_PRIMARY_ERROR_KEY, hash, real_toInt(GameTimer_currentTime))
				call Log_error("|cffFF3A29Error:|r " + msg + "")
			endif
		endif
		call I2S(1 / 0)
	endif
endfunction

function call_doPeriodicallyCounted_ClosureTimers takes integer this_1, integer cb_1 returns nothing
	call int_assertEquals(dispatch_CallbackCounted_CallbackCounted_getCount(cb_1), w_count[this_1])
	if dispatch_CallbackCounted_CallbackCounted_isLast(cb_1) then
		call real_assertEquals(dispatch_CallbackCounted_CallbackCounted_progress(cb_1), 1., 0.0001)
		call int_assertEquals(dispatch_CallbackCounted_CallbackCounted_getCount(cb_1), 1)
	elseif w_count[this_1] == 3 then
		call real_assertEquals(dispatch_CallbackCounted_CallbackCounted_progress(cb_1), 0.33333, 0.00001)
		call int_assertEquals(dispatch_CallbackCounted_CallbackCounted_getCount(cb_1), 3)
	elseif w_count[this_1] == 2 then
		call real_assertEquals(dispatch_CallbackCounted_CallbackCounted_progress(cb_1), 0.66666, 0.00001)
		call int_assertEquals(dispatch_CallbackCounted_CallbackCounted_getCount(cb_1), 2)
	endif
	set w_count[this_1] = w_count[this_1] - 1
endfunction

function timer_release takes timer this_1 returns nothing
	if this_1 == null then
		call error("Trying to release a null timer")
		return
	endif
	if timer_getData(this_1) == TimerUtils_HELD then
		call error("ReleaseTimer: Double free!")
		return
	endif

function tdHandleArcherSplitArrowDamage takes nothing returns nothing
	local unit source
	local unit target
	local integer level
	local integer remainingTargets
	local real center_x
	local real center_y
	local integer laneId
	local group g
	local group iterator
	local unit victim
	local real tuple_temp
	local real tuple_temp_1
	if DamageEvent_isSpell() or DamageEvent_isCode() then
		return
	endif
	set source = DamageEvent_getSource()
	set target = DamageEvent_getTarget()
	if source == null or target == null or tdTowerChoiceForUnitTypeId(GetUnitTypeId(source)) != TDConfig_TD_TOWER_CHOICE_ARCHER then
		return
	endif
	set level = tdTowerLevelForUnitTypeId(GetUnitTypeId(source))
	set remainingTargets = tdArcherSplitArrowTargetCount(level) - 1
	if remainingTargets <= 0 then
		return
	endif
	set tuple_temp = unit_getPos(source)
	set tuple_temp_1 = unit_getPos_return_y
	set center_x = tuple_temp
	set center_y = tuple_temp_1
	set laneId = tdEnemyLaneId(target)
	set g = CreateGroup()
	call group_enumUnitsInRange(g, center_x, center_y, tdTowerBaseRange(TDConfig_TD_TOWER_CHOICE_ARCHER, level) + tdArcherSplitArrowBounceRange())
	set iterator = group_iterator(g)
	loop
		exitwhen  not group_hasNext(iterator)
		set victim = group_next(iterator)
		if remainingTargets > 0 and victim != target and unit_isAlive(victim) and GetOwningPlayer(victim) == Player(TDConfig_TD_ENEMY_PLAYER_ID) and tdEnemyLaneId(victim) == laneId then
			set remainingTargets = remainingTargets - 1
			call DamageEvent_setNextDamageFromCode()
			call unit_damageTarget(source, victim, DamageEvent_getAmount())
		endif
	endloop
	call group_close(iterator)
	call group_destr(g)
endfunction

function DamageEvent_onDamage takes nothing returns nothing
	local integer dmg_1 = DamageInstance_current
	local integer i
	local integer temp
	local integer listener
	local integer prev
	if  not DamageEvent_abort then
		if DamageInstance_isSpell[dmg_1] then
			call widget_setLife(DamageInstance_target[dmg_1], max_2(DamageInstance_lastDamageHP[dmg_1], 1.00))
			if DamageInstance_lastDamageHP[dmg_1] < 0.405 then
				call disableDamageDetect()
				call UnitDamageTarget(DamageInstance_source[dmg_1], DamageInstance_target[dmg_1], 10000., true, false, DamageEvent_ATTACK_TYPE_UNIVERSAL, DAMAGE_TYPE_UNIVERSAL, null)
				call enableDamageDetect()
			endif
		endif
		if unit_getAbilityLevel(DamageInstance_target[dmg_1], DamageEvent_DAMAGE_ABIL_ID) > 0 then
			call unit_removeAbility(DamageInstance_target[dmg_1], DamageEvent_DAMAGE_ABIL_ID)
			call widget_setLife(DamageInstance_target[dmg_1], DamageInstance_lastDamageHP[dmg_1])
		endif
		call dispatch_DamageInstance_DamageInstance_setReducedAmount(dmg_1, DamageInstance_amount[dmg_1])
		set i = 0
		set temp = DamageEvent_maxPriority
		loop
			exitwhen i > temp
			set listener = DamageEvent_firstListeners[i]
			loop
				exitwhen  not (listener != 0)
				set prev = DamageInstance_current
				set DamageInstance_current = dmg_1
				call dispatch_DamageListener_DamageListener_onEvent(listener)
				set DamageInstance_current = prev
				if DamageEvent_abort then
					set DamageInstance_amount[dmg_1] = 0.
					exitwhen true
				else
					set listener = DamageListener_next[listener]
				endif
			endloop
			if DamageEvent_abort then
				exitwhen true
			endif
			set i = i + 1
		endloop
	endif
	call dispatch_DamageInstance_destroyDamageInstance(dmg_1)
	set DamageEvent_recursion = DamageEvent_recursion - 1
	set DamageEvent_abort = false
endfunction

function DamageEvent_onUnreducedDamage takes nothing returns nothing
	local real amount = GetEventDamage()
	local boolean isSpell = false
	local boolean isCode = false
	local unit target
	local integer dmg_1
	local integer i
	local integer temp
	local integer listener
	local integer prev
	local real life
	local real maxHP
	local trigger receiver
	local real prevLife
	local integer clVar
	if DamageEvent_nextDamageType == 2 then
		set isCode = true
	endif
	if amount < 0. then
		set amount = amount * -1.
		set isSpell = true
		set DamageEvent_nextDamageType = 1
	else
		set DamageEvent_nextDamageType = 0
		set DamageEvent_nextDamageElement = DamageEvent_DAMAGE_ELEMENT_ATTACK
	endif
	set target = GetTriggerUnit()
	set dmg_1 = new_DamageInstance(DamageEvent_nextDamageId, GetEventDamageSource(), target, amount, isSpell, isCode, DamageEvent_nextDamageType, DamageEvent_nextDamageElement)
	set DamageEvent_nextDamageId = 0
	set DamageEvent_nextDamageType = 4
	set DamageEvent_nextDamageElement = 0
	set DamageEvent_recursion = DamageEvent_recursion + 1
	if DamageEvent_recursion > DamageEvent_LIMIT_RECURSION then
		call error("There is an infinite loop with the Damage Engine")
	endif
	set i = 0
	set temp = DamageEvent_maxUnreducedPriority
	loop
		exitwhen i > temp
		set listener = DamageEvent_firstUnreducedListeners[i]
		loop
			exitwhen  not (listener != 0)
			set prev = DamageInstance_current
			set DamageInstance_current = dmg_1
			call dispatch_DamageListener_DamageListener_onEvent(listener)
			set DamageInstance_current = prev
			if DamageEvent_abort then
				set DamageInstance_amount[dmg_1] = 0.
				exitwhen true
			else
				set listener = DamageListener_next[listener]
			endif
		endloop
		if DamageEvent_abort then
			exitwhen true
		endif
		set i = i + 1
	endloop
	if amount == 0. then
		call dispatch_DamageInstance_destroyDamageInstance(dmg_1)
		set DamageEvent_recursion = DamageEvent_recursion - 1
		return
	endif
	set life = widget_getLife(target)
	set maxHP = unit_getMaxHP(target)
	if DamageInstance_isSpell[dmg_1] then
		if DamageEvent_dmgTrg == null then
			set receiver = CreateTrigger()
			call trigger_addAction(receiver, ref_function_DamageEvent_onTrg)
			set DamageEvent_dmgTrg = receiver
		endif
		call dispatch_LinkedList_LinkedList_add_1(DamageEvent_damageList, dmg_1)
		set prevLife = life
		if life + amount * 0.75 > maxHP then
			set life = max_2(maxHP - amount / 2.00, 1.00)
			call widget_setLife(target, life)
			set life = (life + maxHP) / 2.00
		else
			set life = life + amount * 0.50
		endif
		set DamageInstance_lastDamageHP[dmg_1] = prevLife - DamageInstance_amount[dmg_1]
		call trigger_registerUnitStateEvent(DamageEvent_dmgTrg, target, UNIT_STATE_LIFE, GREATER_THAN, life)
	else
		if DamageInstance_amount[dmg_1] != amount then
			set life = life + (amount - DamageInstance_amount[dmg_1])
			if maxHP < life then
				set DamageInstance_lastDamageHP[dmg_1] = life - amount
				call unit_addAbility(target, DamageEvent_DAMAGE_ABIL_ID)
			endif
			call widget_setLife(target, max_2(life, 0.42))
		endif
		set clVar = alloc_CallbackSingle_nullTimer_DamageEvent_DamageEvent()
		call construct_CallbackSingle(clVar)
		set dmg[clVar] = dmg_1
		call nullTimer(clVar)
	endif
endfunction

	local integer wave_5
	if enemy == null or tdIsGameOver() then
		return
	endif
	if GetOwningPlayer(enemy) != Player(TDConfig_TD_ENEMY_PLAYER_ID) then
		return
	endif
	set playerId_3 = tdEnemyLaneId(enemy)
	if ( not tdIsLaneActive(playerId_3)) or tdEnemyWasLeaked(enemy) then
		return
	endif
	set wave_5 = tdEnemyWave(enemy)
	call tdClearEnemyHealthTag(enemy)
	call tdGainGold(playerId_3, tdWaveKillGold(wave_5))
	call tdAddScore(playerId_3, tdWaveScoreValue(wave_5))
	call tdRefreshHud()
endfunction

function tdHandleEnemyLeak takes unit enemy returns nothing
	local integer playerId_3
	local integer wave_5
	if enemy == null or ( not unit_isAlive(enemy)) then
		return
	endif
	set playerId_3 = tdEnemyLaneId(enemy)
	set wave_5 = tdEnemyWave(enemy)
	if ( not tdIsLaneActive(playerId_3)) or tdIsLaneDefeated(playerId_3) then
		return
	endif
	call tdLoseLife(playerId_3, 1)
	call tdMarkEnemyLeaked(enemy)
	call tdClearEnemyHealthTag(enemy)
	call unit_kill(enemy)
	if tdGetLives(playerId_3) <= 0 then
		call tdSetResolvedWave(playerId_3, wave_5)
		call tdSetLaneDefeated(playerId_3, true)
		call tdClearLaneEnemies(playerId_3)
		call tdBroadcast("玩家" + int_toString(playerId_3 + 1) + "的路线失守了。")
	endif
	call tdRefreshHud()
	call tdHandleGameResolved()
	call tdTryScheduleNextWave(wave_5)
endfunction

function tdApplyPerfectUpgrade takes unit tower, integer towerChoice returns nothing
	local integer baseAbilityId = tdTowerBaseSkillAbilityId(towerChoice)
	local integer perfectAbilityId = tdTowerPerfectSkillAbilityId(towerChoice)
	if perfectAbilityId == 0 or unit_getAbilityLevel(tower, perfectAbilityId) > 0 then
		return
	endif
	if baseAbilityId > 0 then
		call unit_removeAbility(tower, baseAbilityId)
	endif
	call unit_addPermanentAbility(tower, perfectAbilityId, 1)
	call unit_setScale(tower, tdPerfectUpgradeScaleMultiplier())
	call widget_flashEffect(tower, tdPerfectUpgradeEffectPath(towerChoice), "origin")
	call tdAttachPerfectUpgradeVisuals(tower)
	call DisplayTimedTextToPlayer(GetOwningPlayer(tower), 0., 0., 4.0, tdPerfectUpgradeMessage(towerChoice))
endfunction

function tdEnemyHealthPips takes real ratio returns string
	local integer filled = R2I(ratio * 8.0 + 0.5)
	local string result_1
	local integer i
	if filled < 0 then
		set filled = 0
	endif
	if filled > 8 then
		set filled = 8
	endif
	set result_1 = "["
	set i = 1
	loop
		exitwhen i > 8
		if i <= filled then
			set result_1 = result_1 + "|"
		else
			set result_1 = result_1 + "."
		endif
		set i = i + 1
	endloop
	set result_1 = result_1 + "]"
	return result_1
endfunction

function tdShowEnemyHealthEnum takes nothing returns nothing
	local unit enemy = GetEnumUnit()
	local real maxLife
	local real ratio
	local texttag tag
	if enemy == null or ( not unit_isAlive(enemy)) or GetOwningPlayer(enemy) != Player(TDConfig_TD_ENEMY_PLAYER_ID) then
		return
	endif
	set maxLife = GetUnitState(enemy, UNIT_STATE_MAX_LIFE)
	if maxLife <= 0.405 then
		return
	endif
	set ratio = widget_getLife(enemy) / maxLife
	set tag = tdEnemyHealthTag(enemy)
	call texttag_setText(tag, tdEnemyHealthPips(ratio), 7.0)
	call texttag_setPos(tag, vec2_withZ(unit_getPos(enemy), unit_getPos_return_y, 135.), vec2_withZ_return_y, vec2_withZ_return_z)
	if ratio >= 0.66 then
		call texttag_setColor(tag, 80, 255, 80, 255)
	elseif ratio >= 0.33 then
		call texttag_setColor(tag, 255, 220, 60, 255)
	else
		call texttag_setColor(tag, 255, 90, 90, 255)
	endif
	call texttag_matchVisibility(tag, enemy)
endfunction

function tdEnemyReachedWaypointRect takes unit enemy, rect waypointRect returns boolean
	local real tuple_temp = unit_getPos(enemy)
	local real tuple_temp_1 = unit_getPos_return_y
	local real position_x = tuple_temp
	local real position_y = tuple_temp_1
	return position_x >= GetRectMinX(waypointRect) and position_x <= GetRectMaxX(waypointRect) and position_y >= GetRectMinY(waypointRect) and position_y <= GetRectMaxY(waypointRect)
endfunction

function tdAdvanceEnemyWaypoint takes unit enemy returns nothing
	local integer playerId_3
	local integer waypointIndex
	local rect waypointRect
	if enemy == null or ( not unit_isAlive(enemy)) or GetOwningPlayer(enemy) != Player(TDConfig_TD_ENEMY_PLAYER_ID) or tdEnemyWasLeaked(enemy) then
		return
	endif
	set playerId_3 = tdEnemyLaneId(enemy)
	set waypointIndex = tdEnemyWaypointIndex(enemy)
	if waypointIndex >= tdLaneWaypointCount(playerId_3) then
		return
	endif
	set waypointRect = tdLaneWaypointRect(playerId_3, waypointIndex)
	if  not tdEnemyReachedWaypointRect(enemy, waypointRect) then
		return
	endif
	call tdSetEnemyWaypointIndex(enemy, waypointIndex + 1)
	call tdOrderEnemyAlongLane(enemy)
endfunction

function InitCustomPlayerSlots takes nothing returns nothing
	call SetPlayerStartLocation(Player(0), 0)
	call SetPlayerColor(Player(0), ConvertPlayerColor(0))
	call SetPlayerRacePreference(Player(0), RACE_PREF_HUMAN)
	call SetPlayerRaceSelectable(Player(0), false)
	call SetPlayerController(Player(0), MAP_CONTROL_USER)
	call SetPlayerStartLocation(Player(1), 1)
	call SetPlayerColor(Player(1), ConvertPlayerColor(1))
	call SetPlayerRacePreference(Player(1), RACE_PREF_RANDOM)
	call SetPlayerRaceSelectable(Player(1), false)
	call SetPlayerController(Player(1), MAP_CONTROL_COMPUTER)
endfunction

function InitCustomTeams takes nothing returns nothing
	call SetPlayerTeam(Player(0), 0)
	call SetPlayerTeam(Player(1), 1)
endfunction

function config takes nothing returns nothing
	call SetMapName("TRIGSTR_008")
	call SetMapDescription("")
	call SetPlayers(2)
	call SetTeams(2)
	call SetGamePlacement(MAP_PLACEMENT_USE_MAP_SETTINGS)
	call DefineStartLocation(0, 0.0,  - 192.0)
	call DefineStartLocation(1, 0.0,  - 192.0)
	call InitCustomPlayerSlots()
	call InitCustomTeams()
endfunction
